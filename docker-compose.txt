version: "3.8"

services:
  cyberguardian:
    container_name: cyberguardian
    build:
      context: ./cyberguardian
      dockerfile: Dockerfile
    volumes:
      - ./cyberguardian:/app
      - cyberguardian_node_modules:/app/node_modules
      - cyberguardian_next:/app/.next
    environment:
      - NODE_ENV=development
    networks:
      - nginxproxymanager_default
    restart: unless-stopped

  cybertoolsServer:
    container_name: cybertoolsServer
    build:
      context: ./cybertoolsServer
      dockerfile: Dockerfile
    volumes:
      - ./cybertoolsServer:/app
    networks:
      - nginxproxymanager_default
      - cybertools_network
    restart: unless-stopped
    depends_on:
      - db

  db:
    image: mongo:latest
    volumes:
      - cybertoolsServer:/data/db
    networks:
      - cybertools_network
    restart: unless-stopped

networks:
  nginxproxymanager_default:
    external: true
  cybertools_network:
    driver: bridge

volumes:
  cyberguardian_node_modules:
  cyberguardian_next:
  cybertoolsServer:
